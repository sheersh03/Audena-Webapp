create_agent.py 
import audena
client = audena.Client(api_key="aud_live_...")
# Define a new voice AI agent
agent = client.agents.create(
    name="Customer Support Agent",
    voice_id="claire_hd_v2",
    prompt="You are a helpful assistant...",
    capabilities=[
        "realtime_transcription",
        "emotional_intelligence"
    ]
)
# Webhook for call events
agent.set_webhook("https://api.acme.com/v1/calls")
print(f"Agent {agent.id} is ready.")
